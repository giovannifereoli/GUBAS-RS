13/// Q parameter as defined by Hou 2016.
14pub fn q_ijk(i: f64, j: f64, k: f64) -> f64 {
15    ifact(i as usize) * ifact(j as usize) * ifact(k as usize)
16        / ifact((i + j + k + 3.0) as usize)
17}

19// ── tet_sums ──────────────────────────────────────────────────────────────────

21/// Summation over a tetrahedron with one vertex at the origin.
22/// Vertices are at coordinates (x1,y1,z1), (x2,y2,z2), (x3,y3,z3) in km.
23///
24/// Mirrors `double tet_sums(...)` in C++.
25#[allow(clippy::too_many_arguments)]
26pub fn tet_sums(l: f64, m: f64, n: f64,
27                x1: f64, x2: f64, x3: f64,
28                y1: f64, y2: f64, y3: f64,
29                z1: f64, z2: f64, z3: f64) -> f64 {
30    let mut s = 0.0_f64;
31    let li = l as usize;
32    let mi = m as usize;
33    let ni = n as usize;
34    let lf = ifact(li);
35    let mf = ifact(mi);
36    let nf = ifact(ni);
37    for i1 in 0..=li {
38        for j1 in 0..=(li - i1) {
39            let c_l = lf / (ifact(i1) * ifact(j1) * ifact(li - i1 - j1));
40            for i2 in 0..=mi {
41                for j2 in 0..=(mi - i2) {
42                    let c_m = mf / (ifact(i2) * ifact(j2) * ifact(mi - i2 - j2));
43                    for i3 in 0..=ni {
44                        for j3 in 0..=(ni - i3) {
45                            let c_n = nf / (ifact(i3) * ifact(j3) * ifact(ni - i3 - j3));
46                            s += c_l * c_m * c_n
47                                * x1.powi(i1 as i32) * x2.powi(j1 as i32)
48                                    * x3.powi((li - i1 - j1) as i32)
49                                * y1.powi(i2 as i32) * y2.powi(j2 as i32)
50                                    * y3.powi((mi - i2 - j2) as i32)
51                                * z1.powi(i3 as i32) * z2.powi(j3 as i32)
52                                    * z3.powi((ni - i3 - j3) as i32)
53                                * q_ijk((i1 + i2 + i3) as f64,
54                                        (j1 + j2 + j3) as f64,
55                                        (li + mi + ni - i1 - i2 - i3 - j1 - j2 - j3) as f64);
56                        }
57                    }
58                }
59            }
60        }
61    }
62    s
63}

67fn read_csv_matrix(path: &str) -> Vec<Vec<f64>> {
68    std::fs::read_to_string(path)
69        .unwrap_or_else(|e| panic!("Cannot read {}: {}", path, e))
70        .lines()
71        .filter(|l| !l.trim().is_empty())
72        .map(|line| {
73            line.split(',')
74                .map(|s| s.trim().parse::<f64>().unwrap_or(f64::NAN))
75                .collect::<Vec<f64>>()
76        })
77        .collect()
78}

80/// Load a tetrahedra file and return 0-indexed face data.
81/// Drops all-NaN columns (replicates the Python logic).
82fn load_tet(path: &str) -> Vec<Vec<usize>> {
83    let raw = read_csv_matrix(path);
84    raw.iter()
85        .map(|row| row.iter().filter(|v| !v.is_nan()).map(|&v| (v - 1.0) as usize).collect())
86        .collect()
87}
88
89fn load_vert(path: &str, scale_to_km: bool) -> Vec<Vec<f64>> {
90    let raw = read_csv_matrix(path);
91    if scale_to_km {
92        raw.iter().map(|row| row.iter().map(|&v| v / 1000.0).collect()).collect()
93    } else {
94        raw
95    }
96}

100/// Compute inertia integrals from polyhedron files given in **metres**.
101/// Converts to km internally.  Returns a Cube of size (q+1)^3.
102///
103/// Mirrors `void poly_inertia_met(int q, ...)`.

105    let tet = load_tet(tet_file);
106    let vert = load_vert(vert_file, true); // metres → km

110/// Compute inertia integrals from polyhedron files given in **km**.
111///
112/// Mirrors `void poly_inertia(int q, ...)`.
113pub fn poly_inertia(q: usize, rho: f64, tet_file: &str, vert_file: &str) -> Cube {
114    let tet = load_tet(tet_file);
115    let vert = load_vert(vert_file, false);
116    poly_inertia_impl(q, rho, &tet, &vert)
117}

119fn poly_inertia_impl(q: usize, rho: f64, tet: &[Vec<usize>], vert: &[Vec<f64>]) -> Cube {
120    let mut t = Cube::new(q);
121    for l in 0..=q {
122        for m in 0..=(q - l) {
123            for n in 0..=(q - l - m) {
124                let mut val = 0.0_f64;
125                for face in tet {
126                    // vert row has 4 columns: [index, x, y, z]
127                    let x1 = &vert[face[0]][1..=3];
128                    let x2 = &vert[face[1]][1..=3];
129                    let x3 = &vert[face[2]][1..=3];
130                    // determinant of 3x3 formed by the three vertices
131                    let det_mat: Mat3 = [
132                        [x1[0], x2[0], x3[0]],
133                        [x1[1], x2[1], x3[1]],
134                        [x1[2], x2[2], x3[2]],
135                    ];
136                    let ta_abs = det3(det_mat).abs();
137                    val += rho * ta_abs
138                        * tet_sums(l as f64, m as f64, n as f64,
139                                   x1[0], x2[0], x3[0],
140                                   x1[1], x2[1], x3[1],
141                                   x1[2], x2[2], x3[2]);
142                }
143                t.set(l, m, n, val);
144            }
145        }
146    }
147    t
148}

168fn poly_moi_impl(rho: f64, tet: &[Vec<usize>], vert: &[Vec<f64>]) -> Vec3 {
169    let mut moi = [0.0_f64; 3];
170    for face in tet {
171        let p1: [f64; 3] = [0.0; 3]; // vertex at origin
172        let p2: [f64; 3] = [vert[face[0]][1], vert[face[0]][2], vert[face[0]][3]];
173        let p3: [f64; 3] = [vert[face[1]][1], vert[face[1]][2], vert[face[1]][3]];
174        let p4: [f64; 3] = [vert[face[2]][1], vert[face[2]][2], vert[face[2]][3]];
175        let det_mat: Mat3 = [
176            [p2[0], p3[0], p4[0]],
177            [p2[1], p3[1], p4[1]],
178            [p2[2], p3[2], p4[2]],
179        ];
180        let v_vol = rho * det3(det_mat).abs() / 6.0;
181        // Ixx: integral of (y^2 + z^2) dm
182        moi[0] += v_vol * (p1[1].powi(2) + p1[1]*p2[1] + p2[1].powi(2)
183                           + p1[1]*p3[1] + p2[1]*p3[1] + p3[1].powi(2)
184                           + p1[2].powi(2) + p1[2]*p2[2] + p2[2].powi(2)
185                           + p1[2]*p3[2] + p2[2]*p3[2] + p3[2].powi(2)
186                           + p1[1]*p4[1] + p2[1]*p4[1] + p3[1]*p4[1] + p4[1].powi(2)
187                           + p1[2]*p4[2] + p2[2]*p4[2] + p3[2]*p4[2] + p4[2].powi(2)) / 10.0;
188        // Iyy: integral of (x^2 + z^2) dm
189        moi[1] += v_vol * (p1[0].powi(2) + p1[0]*p2[0] + p2[0].powi(2)
190                           + p1[0]*p3[0] + p2[0]*p3[0] + p3[0].powi(2)
191                           + p1[2].powi(2) + p1[2]*p2[2] + p2[2].powi(2)
192                           + p1[2]*p3[2] + p2[2]*p3[2] + p3[2].powi(2)
193                           + p1[0]*p4[0] + p2[0]*p4[0] + p3[0]*p4[0] + p4[0].powi(2)
194                           + p1[2]*p4[2] + p2[2]*p4[2] + p3[2]*p4[2] + p4[2].powi(2)) / 10.0;
195        // Izz: integral of (x^2 + y^2) dm
196        moi[2] += v_vol * (p1[1].powi(2) + p1[1]*p2[1] + p2[1].powi(2)
197                           + p1[1]*p3[1] + p2[1]*p3[1] + p3[1].powi(2)
198                           + p1[0].powi(2) + p1[0]*p2[0] + p2[0].powi(2)
199                           + p1[0]*p3[0] + p2[0]*p3[0] + p3[0].powi(2)
200                           + p1[1]*p4[1] + p2[1]*p4[1] + p3[1]*p4[1] + p4[1].powi(2)
201                           + p1[0]*p4[0] + p2[0]*p4[0] + p3[0]*p4[0] + p4[0].powi(2)) / 10.0;
202    }
203    moi
204}

206// ── ell_mass_params_met ───────────────────────────────────────────────────────
207
208/// Ellipsoidal inertia integrals and moments, with semi-axes in **metres**.
209///
210/// Returns `(I, T)` where `I = [Ixx, Iyy, Izz]` and `T` is a Cube of size
211/// `(order+1)^3`.  Mirrors `void ell_mass_params_met(...)`.
212pub fn ell_mass_params_met(order: usize, order_body: usize,
213                           rho: f64, a: f64, b: f64, c: f64)
214                           -> (Vec3, Cube) {
215    let a = a / 1000.0; // metres → km
216    let b = b / 1000.0;
217    let c = c / 1000.0;
218    let mass = 4.0 * rho * std::f64::consts::PI * a * b * c / 3.0;
219    let mut t = Cube::new(order);
220    t.set(0, 0, 0, mass);
221    let ia = [mass * (b * b + c * c) / 5.0,
222              mass * (a * a + c * c) / 5.0,
223              mass * (b * b + a * a) / 5.0];
224    if order_body > 0 {
225        t.set(2, 0, 0, mass * a * a / 5.0);
226        t.set(0, 2, 0, mass * b * b / 5.0);
227        t.set(0, 0, 2, mass * c * c / 5.0);
228        if order_body > 3 {
229            t.set(4, 0, 0, 3.0 * mass * a.powi(4) / 35.0);
230            t.set(0, 4, 0, 3.0 * mass * b.powi(4) / 35.0);
231            t.set(0, 0, 4, 3.0 * mass * c.powi(4) / 35.0);
232            t.set(2, 2, 0, mass * a * a * b * b / 35.0);
233            t.set(0, 2, 2, mass * c * c * b * b / 35.0);
234            t.set(2, 0, 2, mass * a * a * c * c / 35.0);
235        }
236    }
237    (ia, t)
238}

240// ── inertia_rot ───────────────────────────────────────────────────────────────
241
242/// Rotate inertia integrals from body-fixed frame to A frame using rotation
243/// matrix C (maps from B to A).  Returns a new Cube `Tp`.
244///
245/// Generic over `T` so dual-number types can flow through for auto-diff.
246/// Mirrors `void inertia_rot(mat C, int q, cube* T, cube* Tp)`.
247pub fn inertia_rot<T: Scalar>(c: Mat3<T>, q: usize, t: &Cube<T>) -> Cube<T> {
248    let mut tp = Cube::new(q);
249    for l in 0..=q {
250        for m in 0..=(q - l) {
251            for n in 0..=(q - l - m) {
252                let mut val = T::zero();
253                for i1 in 0..=l {
254                    for j1 in 0..=(l - i1) {
255                        let cl = ifact(l) / (ifact(i1) * ifact(j1) * ifact(l - i1 - j1));
256                        for i2 in 0..=m {
257                            for j2 in 0..=(m - i2) {
258                                let cm = ifact(m) / (ifact(i2) * ifact(j2) * ifact(m - i2 - j2));
259                                for i3 in 0..=n {
260                                    for j3 in 0..=(n - i3) {
261                                        let cn = ifact(n) / (ifact(i3) * ifact(j3) * ifact(n - i3 - j3));
262                                        let sum_i = i1 + i2 + i3;
263                                        let sum_j = j1 + j2 + j3;
264                                        let sum_k = l + m + n - sum_i - sum_j;
265                                        if sum_i <= q && sum_j <= q && sum_k <= q {
266                                            let coeff = <T as NumCast>::from(cl * cm * cn).unwrap();
267                                            val += coeff
268                                                * c[0][0].powi(i1 as i32) * c[0][1].powi(j1 as i32)
269                                                  * c[0][2].powi((l - i1 - j1) as i32)
270                                                * c[1][0].powi(i2 as i32) * c[1][1].powi(j2 as i32)
271                                                  * c[1][2].powi((m - i2 - j2) as i32)
272                                                * c[2][0].powi(i3 as i32) * c[2][1].powi(j3 as i32)
273                                                  * c[2][2].powi((n - i3 - j3) as i32)
274                                                * t.get(sum_i, sum_j, sum_k);
275                                        }
276                                    }
277                                }
278                            }
279                        }
280                    }
281                }
282                tp.set(l, m, n, val);
283            }
284        }
285    }
286    tp
287}

289// ── dT_dc ─────────────────────────────────────────────────────────────────────
290
291/// Partial of the rotated inertia integral Cube with respect to rotation
292/// matrix element C(row, col).
293///
294/// Generic over `T` for auto-diff.
295/// Mirrors `void dT_dc(int i, int j, mat C, int q, cube* TA, cube* dT)`.
296pub fn dt_dc<T: Scalar>(row: usize, col: usize, c: Mat3<T>, q: usize, t: &Cube<T>) -> Cube<T> {
297    let mut dt = Cube::new(q);
298    for l in 0..=q {
299        for m in 0..=(q - l) {
300            for n in 0..=(q - l - m) {
301                let mut val = T::zero();
302                for i1 in 0..=l {
303                    for j1 in 0..=(l - i1) {
304                        let cl = ifact(l) / (ifact(i1) * ifact(j1) * ifact(l - i1 - j1));
305                        for i2 in 0..=m {
306                            for j2 in 0..=(m - i2) {
307                                let cm = ifact(m) / (ifact(i2) * ifact(j2) * ifact(m - i2 - j2));
308                                for i3 in 0..=n {
309                                    for j3 in 0..=(n - i3) {
310                                        let cn = ifact(n) / (ifact(i3) * ifact(j3) * ifact(n - i3 - j3));
311                                        let si = i1 + i2 + i3;
312                                        let sj = j1 + j2 + j3;
313                                        let sk = l + m + n - si - sj;
314                                        if si <= q && sj <= q && sk <= q {
315                                            let cv = rotation_partial_coeff(
316                                                row, col, c,
317                                                i1, j1, l - i1 - j1,
318                                                i2, j2, m - i2 - j2,
319                                                i3, j3, n - i3 - j3,
320                                            );
321                                            let coeff = <T as NumCast>::from(cl * cm * cn).unwrap();
322                                            val += coeff * cv * t.get(si, sj, sk);
323                                        }
324                                    }
325                                }
326                            }
327                        }
328                    }
329                }
330                dt.set(l, m, n, val);
331            }
332        }
333    }
334    dt
335}

337/// Compute `exp * C[row][col]^(exp-1) * product-of-all-other-C-elements`.
338///
339/// This is the power-rule partial for the monomial of rotation matrix elements
340/// that appears in each term of `inertia_rot`.  Computing it without division
341/// by `C[row][col]` keeps the function correct for dual-number types and when
342/// a matrix element is exactly zero.
343fn rotation_partial_coeff<T: Scalar>(row: usize, col: usize, c: Mat3<T>,
344                                     i1: usize, j1: usize, k1: usize,
345                                     i2: usize, j2: usize, k2: usize,
346                                     i3: usize, j3: usize, k3: usize) -> T {
347    let exp = match (row, col) {
348        (0, 0) => i1, (0, 1) => j1, (0, 2) => k1,
349        (1, 0) => i2, (1, 1) => j2, (1, 2) => k2,
350        (2, 0) => i3, (2, 1) => j3, (2, 2) => k3,
351        _ => unreachable!(),
352    };
353    if exp == 0 { return T::zero(); }
354    // Reduce the exponent of c[row][col] by 1 (power rule), leave others unchanged.
355    let ei1 = if (row, col) == (0, 0) { i1 - 1 } else { i1 };
356    let ej1 = if (row, col) == (0, 1) { j1 - 1 } else { j1 };
357    let ek1 = if (row, col) == (0, 2) { k1 - 1 } else { k1 };
358    let ei2 = if (row, col) == (1, 0) { i2 - 1 } else { i2 };
359    let ej2 = if (row, col) == (1, 1) { j2 - 1 } else { j2 };
360    let ek2 = if (row, col) == (1, 2) { k2 - 1 } else { k2 };
361    let ei3 = if (row, col) == (2, 0) { i3 - 1 } else { i3 };
362    let ej3 = if (row, col) == (2, 1) { j3 - 1 } else { j3 };
363    let ek3 = if (row, col) == (2, 2) { k3 - 1 } else { k3 };
364    let prod = c[0][0].powi(ei1 as i32) * c[0][1].powi(ej1 as i32) * c[0][2].powi(ek1 as i32)
365             * c[1][0].powi(ei2 as i32) * c[1][1].powi(ej2 as i32) * c[1][2].powi(ek2 as i32)
366             * c[2][0].powi(ei3 as i32) * c[2][1].powi(ej3 as i32) * c[2][2].powi(ek3 as i32);
367    <T as NumCast>::from(exp as f64).unwrap() * prod
368}
