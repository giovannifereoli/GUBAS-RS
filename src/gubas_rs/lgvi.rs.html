1//! Lie Group Variational Integrator (LGVI) helpers (Lee 2007).
2//!
3//! Implements the Hamiltonian map used by `lgvi_integ`.  All intermediate
4//! computations use dimensionless normalised variables; physical inputs/outputs
5//! are in km / kg / s.
6//!
7//! Rotation updates are solved implicitly with two Newton solvers:
8//! - Cayley parametrisation (`f_cayley_calc`) — primary solver
9//! - Rodrigues / exponential-map fallback (`f_exp_calc_scaled`)
10//!
11//! # Normalisation scales
12//! Given initial state `x0` with `|r0| = nr`:
13//! - Length scale: `nr` (km)
14//! - Mass scale:   `nm = Ma + Mb` (kg)
15//! - Time scale:   `nt = sqrt(G·nm / nr³)` (1/s)
16
17use crate::inertia::{dt_dc, inertia_rot};

19use crate::potential::{du_c, du_x};

24/// Outer (tensor) product  a ⊗ b  as a 3×3 matrix.
25fn outer(a: Vec3, b: Vec3) -> Mat3 {
26    [[a[0]*b[0], a[0]*b[1], a[0]*b[2]],
27     [a[1]*b[0], a[1]*b[1], a[1]*b[2]],
28     [a[2]*b[0], a[2]*b[1], a[2]*b[2]]]
29}

31// ── map_potential_partials_lgvi ───────────────────────────────────────────────
32
33/// Compute the mutual-potential partial du/dr (Vec3) and the gravity torque M
34/// (Vec3) on the secondary body in the A frame, given physical position r and
35/// relative rotation C.
36///
37/// Mirrors `void map_potential_partials(mat* C, mat* r, ...)`.
38pub fn map_potential_partials_lgvi(c: Mat3, r: Vec3, p: &Params) -> (Vec3, Vec3) {
39    let r_mag = norm(r).max(1e-30);
40    let e     = scale_v(1.0 / r_mag, r);
41
42    let tbp = inertia_rot(c, p.n, &p.tb);
43
44    let dts: [[Cube; 3]; 3] = std::array::from_fn(|i| {
45        std::array::from_fn(|j| dt_dc(i, j, c, p.n, &p.tb))
46    });
47
48    let du_dr: Vec3 = [
49        du_x(p.g, p.n, &p.tk, &p.a, &p.b, e, r_mag, 0, &p.ta, &tbp),
50        du_x(p.g, p.n, &p.tk, &p.a, &p.b, e, r_mag, 1, &p.ta, &tbp),
51        du_x(p.g, p.n, &p.tk, &p.a, &p.b, e, r_mag, 2, &p.ta, &tbp),
52    ];
53
54    let du_cols: [Vec3; 3] = std::array::from_fn(|col| {
55        [
56            du_c(p.g, p.n, &p.tk, &p.a, &p.b, e, r_mag, &p.ta, &dts[0][col]),
57            du_c(p.g, p.n, &p.tk, &p.a, &p.b, e, r_mag, &p.ta, &dts[1][col]),
58            du_c(p.g, p.n, &p.tk, &p.a, &p.b, e, r_mag, &p.ta, &dts[2][col]),
59        ]
60    });
61
62    // M = Σ_col  cross(C[:,col], du/dC[:,col])  — torque on secondary (A frame)
63    let m_grav = add_v(add_v(
64        cross(col(c, 0), du_cols[0]),
65        cross(col(c, 1), du_cols[1])),
66        cross(col(c, 2), du_cols[2]));
67
68    (du_dr, m_grav)
69}

71// ── F_exp_calc (Rodrigues / exponential-map implicit solver) ──────────────────
72
73/// Rodrigues implicit solver used as fallback when the Cayley solver diverges.
74///
75/// `g_scaled` is already multiplied by the step size h (because it is called
76/// from within `f_cayley_calc` after g was pre-scaled there, matching the C++
77/// calling convention).
78///
79/// Returns (f, F) where F is the corresponding SO(3) rotation matrix.
80fn f_exp_calc_scaled(g_scaled: Vec3, i_mat: Mat3) -> (Vec3, Mat3) {
81    let mut f: Vec3 = [0.1; 3];
82
83    for iter in 0..101 {
84        let fn_ = norm(f).max(1e-300);
85        let i_f = mat_vec(i_mat, f);
86        let sin_fn = fn_.sin();
87        let cos_fn = fn_.cos();
88
89        // G(f) = sin|f|/|f| · I·f  +  (1−cos|f|)/|f|² · (f × I·f)
90        let big_g = add_v(
91            scale_v(sin_fn / fn_,                  i_f),
92            scale_v((1.0 - cos_fn) / (fn_ * fn_),  cross(f, i_f)),
93        );
94
95        let residual = sub_v(g_scaled, big_g);
96        if norm(residual) <= 1e-15 { break; }
97        if iter == 100 {
98            eprintln!("Warning: F_exp_calc did not converge to numerical precision.");
99            break;
100        }
101
102        // Jacobian  d G / d f
103        let a1 = (fn_ * cos_fn - sin_fn) / fn_.powi(3);
104        let a2 = sin_fn / fn_;
105        let a3 = (fn_ * sin_fn - 2.0 * (1.0 - cos_fn)) / fn_.powi(4);
106        let a4 = (1.0 - cos_fn) / (fn_ * fn_);
107
108        let grad_g = add_m(
109            add_m(
110                add_m(scale_m(a1, outer(i_f, f)),
111                      scale_m(a2, i_mat)),
112                scale_m(a3, outer(cross(f, i_f), f)),
113            ),
114            // (1−cos|f|)/|f|² · ( −tilde(I·f)  +  tilde(f)·I )
115            scale_m(a4, add_m(
116                scale_m(-1.0, tilde(i_f)),
117                mat_mul(tilde(f), i_mat),
118            )),
119        );
120
121        f = add_v(f, mat_vec(inv3(grad_g), residual));
122    }
123
124    // Rodrigues rotation matrix
125    let fn_ = norm(f).max(1e-300);
126    let t   = tilde(f);
127    let big_f = add_m(
128        add_m(eye(), scale_m(fn_.sin() / fn_,                      t)),
129                     scale_m((1.0 - fn_.cos()) / (fn_ * fn_), mat_mul(t, t)),
130    );
131    (f, big_f)
132}

134// ── F_cayley_calc ─────────────────────────────────────────────────────────────
135
136/// Cayley implicit solver for the LGVI rotation update.
137///
138/// Solves  G(f) ≡ g + g×f + (g·f)f − 2·I·f = 0  via Newton's method,
139/// then returns the Cayley rotation  F = (I+tilde(f))·(I−tilde(f))⁻¹.
140///
141/// Falls back to `f_exp_calc_scaled` after 100 iterations, matching the C++.
142fn f_cayley_calc(h: f64, g_in: Vec3, i_mat: Mat3) -> (Vec3, Mat3) {
143    let mut f: Vec3 = [0.0001; 3];
144    let g = scale_v(h, g_in);      // scale once, matches C++: `(*g)=h*(*g)`
145    let mut check = 0usize;
146
147    loop {
148        // Residual  G = g + cross(g,f) + dot(g,f)·f − 2·I·f
149        let big_g = sub_v(
150            add_v(add_v(g, cross(g, f)), scale_v(dot(g, f), f)),
151            scale_v(2.0, mat_vec(i_mat, f)),
152        );
153
154        if norm(big_g) <= 1e-15 {
155            // Converged — Cayley rotation: F = (I+tilde(f))·(I−tilde(f))⁻¹
156            let big_f = mat_mul(
157                add_m(eye(), tilde(f)),
158                inv3(sub_m(eye(), tilde(f))),
159            );
160            return (f, big_f);
161        }
162
163        // Jacobian  dG/df = tilde(g) + (g·f)·I + outer(f,g) − 2·I_mat
164        let grad_g = sub_m(
165            add_m(add_m(tilde(g), scale_m(dot(g, f), eye())), outer(f, g)),
166            scale_m(2.0, i_mat),
167        );
168        f = sub_v(f, mat_vec(inv3(grad_g), big_g));
169        check += 1;
170
171        if check > 100 {
172            eprintln!("broke Cayley");
173            let cay_err = norm(big_g);
174            let cay_f   = f;
175
176            // Fallback: exponential solver (g is already h-scaled here)
177            let (exp_f, exp_f_mat) = f_exp_calc_scaled(g, i_mat);
178
179            // Cayley residual at exp solution — compare against Cayley's error
180            let exp_big_g = sub_v(
181                add_v(add_v(g, cross(g, exp_f)), scale_v(dot(g, exp_f), exp_f)),
182                scale_v(2.0, mat_vec(i_mat, exp_f)),
183            );
184            if norm(sub_v(g, exp_big_g)) > cay_err {
185                // Exp did not improve; use Cayley f with exp-derived F (mirrors C++)
186                return (cay_f, exp_f_mat);
187            } else {
188                return (exp_f, exp_f_mat);
189            }
190        }
191    }
192}

194// ── hamiltonian_map ───────────────────────────────────────────────────────────
195
196/// One LGVI Hamiltonian map step.
197///
198/// Inputs:
199///   `h`       — time step (s, physical)
200///   `params`  — simulation parameters
201///   `x`       — current state \[r,v,wc,ws,Cc,C\] (30 elements, physical)
202///   `x0`      — initial state used for normalization
203///   `du_dr_n` — previous potential partial ∂U/∂r (physical, km/kg/s²·kg = force/mass … km²/s²/km)
204///   `m_n`     — previous gravity torque on secondary (physical, N·km)
205///
206/// Returns  (x_new, du_dr_n1, m_n1).
207///
208/// Mirrors `void hamiltonian_map(...)`.
209pub fn hamiltonian_map(
210    h:        f64,
211    params:   &Params,
212    x:        [f64; 30],
213    x0:       [f64; 30],
214    du_dr_n:  Vec3,
215    m_n:      Vec3,
216) -> ([f64; 30], Vec3, Vec3) {
217
218    // ── normalisation constants ──────────────────────────────────────────
219    let r0_vec: Vec3 = [x0[0], x0[1], x0[2]];
220    let nr  = norm(r0_vec);                          // length scale  (km)
221    let nm  = params.m;                              // mass scale    (kg)
222    let nt  = (params.g * nm / nr.powi(3)).sqrt();   // time⁻¹ scale  (1/s)
223
224    let alpha = nm * nr * nr;                        // nm·nr²
225    let beta  = nm * nr * nt * nt;                   // nm·nr·nt² = G·nm²/nr²
226
227    let h_n = h * nt;   // normalised step
228
229    // ── unpack and normalise state ───────────────────────────────────────
230    let r : Vec3 = scale_v(1.0 / nr,       [x[0], x[1], x[2]]);
231    let v : Vec3 = scale_v(1.0 / (nr*nt),  [x[3], x[4], x[5]]);
232    let wc: Vec3 = scale_v(1.0 / nt,       [x[6], x[7], x[8]]);
233    let ws: Vec3 = scale_v(1.0 / nt,       [x[9], x[10], x[11]]);
234    let cc: Mat3 = [[x[12], x[13], x[14]],
235                    [x[15], x[16], x[17]],
236                    [x[18], x[19], x[20]]];
237    let c : Mat3 = [[x[21], x[22], x[23]],
238                    [x[24], x[25], x[26]],
239                    [x[27], x[28], x[29]]];
240
241    // Normalised potential partials
242    let du_dr_norm: Vec3 = scale_v(1.0 / beta,            du_dr_n);   // ÷ nm·nr·nt²  (force scale)
243    let m_norm    : Vec3 = scale_v(1.0 / (alpha * nt*nt), m_n);       // ÷ nm·nr²·nt² (torque scale)
244
245    let ia = params.ia;
246    let ib = params.ib;
247
248    // ── inertia tensors (normalised) ─────────────────────────────────────
249    // IH  = diag(IA) / (nm·nr²)
250    let ih  : Mat3 = scale_m(1.0 / alpha, diag(ia));
251    // IBr = C · diag(IB) · Cᵀ / (nm·nr²)
252    let ibr : Mat3 = scale_m(
253        1.0 / alpha,
254        mat_mul(mat_mul(c, diag(ib)), transpose(c)),
255    );
256
257    // ── relative-attitude Cayley solve (secondary body B) ────────────────
258    // g_B = IBr·ws − (h/2)·M_norm
259    let g_b = sub_v(mat_vec(ibr, ws), scale_v(h_n / 2.0, m_norm));
260    let (_, fab) = f_cayley_calc(h_n, g_b, ibr);
261
262    // ── inertial-attitude Cayley solve (primary body A) ──────────────────
263    // g_A = IH·wc + (h/2)·cross(r, du_dr_norm) + (h/2)·M_norm
264    let g_a = add_v(
265        add_v(mat_vec(ih, wc),
266              scale_v(h_n / 2.0, cross(r, du_dr_norm))),
267        scale_v(h_n / 2.0, m_norm),
268    );
269    let (_, fna) = f_cayley_calc(h_n, g_a, ih);
270
271    // ── propagate position and relative attitude ─────────────────────────
272    // r_n1 = Fnaᵀ · (r + h·v − (h²/2)·du_dr_norm)
273    let r_n1: Vec3 = mat_vec(
274        transpose(fna),
275        add_v(add_v(r, scale_v(h_n, v)),
276              scale_v(-h_n * h_n / 2.0, du_dr_norm)),
277    );
278    // C_n1 = Fnaᵀ · Fab · C
279    let c_n1: Mat3 = mat_mul(mat_mul(transpose(fna), fab), c);
280
281    // ── evaluate new potential partials at (C_n1, r_n1·nr) ──────────────
282    let r_n1_phys: Vec3 = scale_v(nr, r_n1);    // de-normalise for the potential call
283    let (du_dr_n1, m_n1) = map_potential_partials_lgvi(c_n1, r_n1_phys, params);
284
285    let du_dr_n1_norm: Vec3 = scale_v(1.0 / beta,            du_dr_n1);
286    let m_n1_norm    : Vec3 = scale_v(1.0 / (alpha * nt*nt), m_n1);
287
288    // ── propagate velocity ───────────────────────────────────────────────
289    // v_n1 = Fnaᵀ·(v − h/2·du_dr_norm) − h/2·du_dr_n1_norm
290    let v_n1: Vec3 = sub_v(
291        mat_vec(transpose(fna), sub_v(v, scale_v(h_n / 2.0, du_dr_norm))),
292        scale_v(h_n / 2.0, du_dr_n1_norm),
293    );
294
295    // ── propagate primary angular velocity ───────────────────────────────
296    // wc_n1 = IH⁻¹ · ( Fnaᵀ·(IH·wc + h/2·cross(r,du_dr_norm) + h/2·M_norm)
297    //                  + h/2·cross(r_n1,du_dr_n1_norm) + h/2·M_n1_norm )
298    // IH⁻¹  = alpha · diag(1/IA)
299    let wc_bracket = add_v(
300        add_v(mat_vec(ih, wc),
301              scale_v(h_n / 2.0, cross(r, du_dr_norm))),
302        scale_v(h_n / 2.0, m_norm),
303    );
304    let wc_rhs = add_v(
305        add_v(mat_vec(transpose(fna), wc_bracket),
306              scale_v(h_n / 2.0, cross(r_n1, du_dr_n1_norm))),
307        scale_v(h_n / 2.0, m_n1_norm),
308    );
309    let ih_inv: Mat3 = scale_m(alpha, diag([1.0/ia[0], 1.0/ia[1], 1.0/ia[2]]));
310    let wc_n1: Vec3  = mat_vec(ih_inv, wc_rhs);
311
312    // ── propagate secondary angular velocity ─────────────────────────────
313    // ws_n1 = C_n1·(IB/α)⁻¹·C_n1ᵀ · ( Fnaᵀ·(IBr·ws − h/2·M_norm) − h/2·M_n1_norm )
314    // (IB/α)⁻¹ expressed in A-frame = α · C_n1·diag(1/IB)·C_n1ᵀ
315    let ws_bracket = sub_v(mat_vec(ibr, ws), scale_v(h_n / 2.0, m_norm));
316    let ws_rhs = sub_v(
317        mat_vec(transpose(fna), ws_bracket),
318        scale_v(h_n / 2.0, m_n1_norm),
319    );
320    let ib_inv_rot: Mat3 = scale_m(
321        alpha,
322        mat_mul(mat_mul(c_n1, diag([1.0/ib[0], 1.0/ib[1], 1.0/ib[2]])), transpose(c_n1)),
323    );
324    let ws_n1: Vec3 = mat_vec(ib_inv_rot, ws_rhs);
325
326    // ── propagate inertial attitude ──────────────────────────────────────
327    // Cc_n1 = Cc · Fna
328    let cc_n1: Mat3 = mat_mul(cc, fna);
329
330    // ── de-normalise and pack output ─────────────────────────────────────
331    let r_out  = r_n1_phys;                   // already physical
332    let v_out  = scale_v(nr * nt, v_n1);
333    let wc_out = scale_v(nt, wc_n1);
334    let ws_out = scale_v(nt, ws_n1);
335
336    let x_out: [f64; 30] = [
337        r_out[0],   r_out[1],   r_out[2],
338        v_out[0],   v_out[1],   v_out[2],
339        wc_out[0],  wc_out[1],  wc_out[2],
340        ws_out[0],  ws_out[1],  ws_out[2],
341        cc_n1[0][0], cc_n1[0][1], cc_n1[0][2],
342        cc_n1[1][0], cc_n1[1][1], cc_n1[1][2],
343        cc_n1[2][0], cc_n1[2][1], cc_n1[2][2],
344        c_n1[0][0],  c_n1[0][1],  c_n1[0][2],
345        c_n1[1][0],  c_n1[1][1],  c_n1[1][2],
346        c_n1[2][0],  c_n1[2][1],  c_n1[2][2],
347    ];
348
349    (x_out, du_dr_n1, m_n1)
350}
